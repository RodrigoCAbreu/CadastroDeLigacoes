Script Banco de Dados

CREATE DATABASE RegistroDeLigacoes;

USE RegistroDeLigacoes;

Create table usuarios (
ID Int UNSIGNED ZEROFILL NOT NULL AUTO_INCREMENT,
nome varchar(50) not null,
login Varchar(30) not null,
senha Varchar(40) not null,
Primary Key (ID)) ENGINE = MyISAM;)


create table cadastro (
ID Int UNSIGNED ZEROFILL NOT NULL AUTO_INCREMENT,
usuario not null,
data date,
hora time,
setor not null,
codigo not null,
prontuario not null,
paciente not null,
consulta,
profissional,
motivo not null,
telefone1 not null,
contato1 not null, 
sit1 not null,
telefone2,
contato2,
sit2,
telefone3,
contato3,
sit3,
telefone4,
contato4,
sit4,
observacoes,
Primary Key (ID)) ENGINE = MyISAM;)
