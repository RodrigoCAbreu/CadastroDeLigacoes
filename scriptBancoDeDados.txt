Script Banco de Dados

CREATE DATABASE RegistroDeLigacoes;

USE RegistroDeLigacoes;

Create table usuarios (
ID Int UNSIGNED ZEROFILL NOT NULL AUTO_INCREMENT,
nome varchar(50) not null,
login Varchar(30) not null,
senha Varchar(40) not null,
Primary Key (ID)) ENGINE = MyISAM;)


create table cadastro (
ID Int UNSIGNED ZEROFILL NOT NULL AUTO_INCREMENT,
usuario varchar(40) not null,
data date,
hora time,
setor varchar(40) not null,
codigo Varchar(40) not null,
prontuario varchar(40) not null,
paciente varchar(40) not null,
consulta varchar(40),
profissional varchar(40),
motivo varchar(150) not null,
telefone1 varchar(12)not null,
contato1 varchar(40) not null, 
sit1 varchar(40) not null,
telefone2 varchar(12),
contato2 varchar(40) ,
sit2 varchar(40),
telefone3 varchar(12),
contato3 varchar(40),
sit3 varchar(40),
telefone4 varchar(12),
contato4 varchar(40),
sit4 varchar(40),
observacoes varchar(250),
Primary Key (ID)) ENGINE = MyISAM;)
